# 多图上传
  ```html
  <!-- 多图上传 -->
  <uni-section title="只选择图片" type="line">
    <view class="example-body">
      <uni-file-picker limit="9" title="最多选择9张图片" @select="select"></uni-file-picker>
    </view>
  </uni-section>
  ```

  ```js
  data() {
    return {
      imageList: []  
    }
  },
  methods: {
    // 图片上传成功时触发
    select(e) {
      this.imageList = e.???
    }
  }
  ```
